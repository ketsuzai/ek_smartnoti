function selectDate(dateStr) {
  selectedDate = dateStr;
  const d = new Date(dateStr);
  // 해당 주 월요일 계산
  const dow = d.getDay(); // 0=일
  const diffToMon = (dow === 0) ? -6 : 1 - dow;
  currentWeekStart = new Date(d);
  currentWeekStart.setDate(d.getDate() + diffToMon);
  currentYear = currentWeekStart.getFullYear();
  currentMonth = currentWeekStart.getMonth();
  viewMode = 'week';

  renderDetailPanel(dateStr);

function renderWeek() {
  document.getElementById('monthView').style.display = 'none';
  document.getElementById('weekView').style.display = 'block';
  document.getElementById('btnToMonth').style.display = 'flex';

  const ws = currentWeekStart;
  const we = new Date(ws); we.setDate(ws.getDate() + 4);
  const ms = ws.getMonth(), me = we.getMonth();
  const title = ms === me
    ? `${ws.getFullYear()}년 ${ws.getMonth()+1}월 ${getWeekNum(ws)}주`
    : `${ws.getMonth()+1}/${ws.getDate()} – ${we.getMonth()+1}/${we.getDate()}`;
  document.getElementById('calTitle').textContent = title;

  const grid = document.getElementById('weekGrid');
  grid.innerHTML = '';
  const dows = ['월','화','수','목','금'];

    const dotsEl = document.createElement('div');

function getWeekNum(d) {
  const firstDay = new Date(d.getFullYear(), d.getMonth(), 1);
  return Math.ceil((d.getDate() + ((firstDay.getDay()+6)%7)) / 7);
}

  modalMode = mode;

  modalClassId = classId;
  modalEditId = meal ? meal.id : null;
  modalImageFile = null;

  const verb = mode === 'add' ? '등록' : '수정';

  document.getElementById('mealModalTitle').textContent = `${mt.label} ${verb}`;
  document.getElementById('mealModalSub').textContent = dateLbl + classLbl;
  document.getElementById('inputMenu').value = meal?.menu || '';
  document.getElementById('inputAllergy').value = meal?.allergy || '';

  // 이미지 미리보기 초기화
  document.getElementById('imagePreview').style.display = 'none';
  document.getElementById('inputImage').value = '';
  if (meal?.imageUrl) {
    document.getElementById('imagePreviewThumb').src = meal.imageUrl;
    document.getElementById('imagePreviewName').textContent = '첨부 이미지';
    document.getElementById('imagePreview').style.display = 'flex';
  }

  document.getElementById('mealModal').classList.add('open');
  setTimeout(() => document.getElementById('inputMenu').focus(), 100);
}

function closeMealModal() {
  document.getElementById('mealModal').classList.remove('open');
  modalImageFile = null;
}

function handleImageSelect(input) {
  const file = input.files[0];
  if (!file) return;
  if (file.size > 5 * 1024 * 1024) { showToast('이미지는 5MB 이하만 가능합니다.', 'danger'); return; }
  modalImageFile = file;
  const reader = new FileReader();
  reader.onload = e => {
    document.getElementById('imagePreviewThumb').src = e.target.result;
    document.getElementById('imagePreviewName').textContent = file.name;
    document.getElementById('imagePreview').style.display = 'flex';
  };
  reader.readAsDataURL(file);
}

function removeImage() {
  modalImageFile = null;
  document.getElementById('inputImage').value = '';
  document.getElementById('imagePreview').style.display = 'none';
  // 수정 시 기존 이미지도 제거 표시
  if (modalEditId) {
    const m = meals.find(x => x.id === modalEditId);
    if (m) m.imageUrl = null;
  }
}

function saveMeal() {
  const menu = document.getElementById('inputMenu').value.trim();
  if (!menu) { showToast('메뉴를 입력해주세요.', 'danger'); document.getElementById('inputMenu').focus(); return; }
  const allergy = document.getElementById('inputAllergy').value.trim();
  const imageUrl = modalImageFile
    ? URL.createObjectURL(modalImageFile)
    : (modalEditId ? meals.find(x => x.id === modalEditId)?.imageUrl || null : null);

  if (modalMode === 'add') {
    meals.push({ id: nextId++, date: modalDate, mealType: modalMealType, classId: modalClassId, menu, allergy, imageUrl });
    showToast('식단이 등록되었습니다.', 'success');
  } else {
    const m = meals.find(x => x.id === modalEditId);
    if (m) { m.menu = menu; m.allergy = allergy; m.imageUrl = imageUrl; }
    showToast('식단이 수정되었습니다.', 'success');
  }
  closeMealModal();
  renderDetailPanel(modalDate);
  // 캘린더 셀 갱신
  if (viewMode === 'month') renderMonth();
  else renderWeek();
}
